CREATE PROC SP_LISTARCLIENTES
AS
SELECT * 
FROM Clientes
WHERE isActiveCli =1
GO


--Insertar en Clientes

CREATE PROC SP_INSERTARCLIENTE
@CODIGO nvarchar (11),
@DNI_RUC nvarchar (10),
@NOMBRES nvarchar (30),
@APELLIDOS nvarchar (50),
@DIRECCION nvarchar (70),
@MANZANA nvarchar (10),
@LOTE int,
@CELULAR nvarchar (15),
@REFERENCIA nvarchar(100)
AS
INSERT INTO Clientes VALUES (@CODIGO,@DNI_RUC,@NOMBRES,@APELLIDOS,@DIRECCION,@MANZANA,@LOTE,@CELULAR,@REFERENCIA,1)

--PROC Almacenado Editar cliente  --Modificado 05/09/2020

CREATE PROC SP_EDITARCLIENTE

@DNI_RUC nvarchar (10),
@NOMBRES nvarchar (30),
@APELLIDOS nvarchar (50),
@DIRECCION nvarchar (70),
@MANZANA nvarchar (10),
@LOTE int,
@CELULAR nvarchar (15),
@REFERENCIA nvarchar(100),
@COD nvarchar (11)
AS
UPDATE Clientes SET DNI_RUC=@DNI_RUC, NOMBRES=@NOMBRES, APELLIDOS=@APELLIDOS, DIRECCION=@DIRECCION, MANZANA=@MANZANA, LOTE=@LOTE,CELULAR=@CELULAR,REFERENCIA=@REFERENCIA 
where CODIGO=@COD


--PROC SP_ELIMINARCLIENTE


CREATE PROC SP_ELIMINARCLIENTE
@CODIGO nvarchar(11)
AS
UPDATE Clientes SET isActiveCli=0
WHERE CODIGO = @CODIGO
